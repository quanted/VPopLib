set (SOURCES

                adult.cpp
                bee.cpp

                weathergriddata.cpp)

                position.cpp
                cuintarray.cpp
                ctime.cpp
                cstringarray.cpp
                cstring.cpp
                cptrlist.cpp
                coledatetime.cpp
                coblist.cpp
                cobject.cpp
                cmapstringtoob.cpp
                ccmdtarget.cpp
                carray.cpp)

                 BASE_NAME libvpop
                 EXPORT_MACRO_NAME libvpop_EXPORT
                 EXPORT_FILE_NAME ${CMAKE_BINARY_DIR}/libvpop_Export.h
                 #STATIC_DEFINE MyLibrary_BUILT_AS_STATIC
            )
